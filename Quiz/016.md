```.py
def stringcount(string):
    temp = ""
    string = list(string.lower())
    for i in range(len(string)):
        if string[i] == " ":
            temp += " "
        else:
            letters = 0
            for j in range(i+1):
                if string[i] == string[j]:
                    letters += 1
            temp += str(letters)

    return temp

l = ["hello world", "aaaaAABB", "abABabAB", "Masamu Oshita"]
for i in l:
    print(i)
    print(stringcount(i)+'\n')
```
