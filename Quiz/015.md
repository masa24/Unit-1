def factorial(a):
    count = 0
    for i in range(1,a+1):
        if a%i == 0:
            count+=1
    return count
door = ''
open = 0
num = int(input('Enter the number'))
for i in range(1,num+1):
    if factorial(i) % 2 == 1:
        door += str(i)
        door += ','
        open += 1
print(f'{open} doors are opened')
print(f'door {door} is oppened')
