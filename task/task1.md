```.py
color =str()
num = int(input())
for a in range(1,num+1):
    if a%4==1:
        color = 'red'
    if a%4==2:
        color = 'white'
    if a%4==3:
        color = 'yellow'
    if a%4==0:
        color = 'blue'
    print(a,color)
```
